Quick Start Examples

cURL 
# Get bridge quotes with multi-hop optimization
curl -X POST http://localhost:8000/api/v1/routes/multi-hop \
  -H "X-API-Key: YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "source_chain": "ethereum",
    "destination_chain": "arbitrum",
    "token": "USDC",
    "amount": "1000000000",
    "max_hops": 2
  }'

# Track any blockchain transaction
curl -X GET http://localhost:8000/api/v1/transactions/track/0xabc123... \
  -H "X-API-Key: YOUR_API_KEY"

# Simulate transaction for testing
curl -X POST http://localhost:8000/api/v1/simulator/simulate \
  -H "X-API-Key: YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "bridge_name": "Across Protocol",
    "source_chain": "ethereum",
    "destination_chain": "arbitrum",
    "amount": "1000000000",
    "completion_time_seconds": 180
  }'

# Get real token prices (FREE APIs)
curl -X GET http://localhost:8000/api/v1/utilities/token-price?symbol=USDC \
  -H "X-API-Key: YOUR_API_KEY"

# Get reliability scores for all bridges
curl -X GET http://localhost:8000/api/v1/analytics/reliability-scores?hours=168 \
  -H "X-API-Key: YOUR_API_KEY"

# Access real-time dashboard
open http://localhost:8000/static/dashboard.html?api_key=YOUR_API_KEY
